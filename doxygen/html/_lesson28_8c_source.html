152 // Calculates 3rd degree polynomial based on array of 4 points
153 // and a single variable (u) which is generally between 0 and 1
154 POINT_3D Bernstein(float u, POINT_3D *p) {
155  POINT_3D a, b, c, d, r;
156 
157  a = pointTimes(pow(u,3), p[0]);
158  b = pointTimes(3*pow(u,2)*(1-u), p[1]);
159  c = pointTimes(3*u*pow((1-u),2), p[2]);
160  d = pointTimes(pow((1-u),3), p[3]);
161 
162  r = pointAdd(pointAdd(a, b), pointAdd(c, d));
163 
164  return r;
165 }

194 // Generates a display list based on the data in the patch
195 // and the number of divisions
196 GLuint genBezier(BEZIER_PATCH patch, int divs) {
197  int u = 0, v;
198  float py, px, pyold;
199  GLuint drawlist = glGenLists(1); // make the display list
200  POINT_3D temp[4];
201  POINT_3D *last = (POINT_3D*)malloc(sizeof(POINT_3D)*(divs+1));
202  // array of points to mark the first line of polys
203 
204  if (patch.dlBPatch != 0) // get rid of any old display lists
205  glDeleteLists(patch.dlBPatch, 1);
206 
207  temp[0] = patch.anchors[0][3]; // the first derived curve (along x axis)
208  temp[1] = patch.anchors[1][3];
209  temp[2] = patch.anchors[2][3];
210  temp[3] = patch.anchors[3][3];
211 
212  for (v=0;v<=divs;v++) { // create the first line of points
213  px = ((float)v)/((float)divs); // percent along y axis
214  // use the 4 points from the derives curve to calculate the points along that curve
215  last[v] = Bernstein(px, temp);
216  }
217 
218  glNewList(drawlist, GL_COMPILE); // Start a new display list
219  glBindTexture(GL_TEXTURE_2D, patch.texture); // Bind the texture
220 
221  for (u=1;u<=divs;u++) {
222  py = ((float)u)/((float)divs); // Percent along Y axis
223  pyold = ((float)u-1.0f)/((float)divs); // Percent along old Y axis
224 
225  temp[0] = Bernstein(py, patch.anchors[0]); // Calculate new bezier points
226  temp[1] = Bernstein(py, patch.anchors[1]);
227  temp[2] = Bernstein(py, patch.anchors[2]);
228  temp[3] = Bernstein(py, patch.anchors[3]);
229 
230  glBegin(GL_TRIANGLE_STRIP); // Begin a new triangle strip
231 
232  for (v=0;v<=divs;v++) {
233  px = ((float)v)/((float)divs); // Percent along the X axis
234 
235  glTexCoord2f(pyold, px); // Apply the old texture coords
236  glVertex3d(last[v].x, last[v].y, last[v].z); // Old Point
237 
238  last[v] = Bernstein(px, temp); // Generate new point
239  glTexCoord2f(py, px); // Apply the new texture coords
240  glVertex3d(last[v].x, last[v].y, last[v].z); // New Point
241  }
242 
243  glEnd(); // END the triangle srip
244  }
245 
246  glEndList(); // END the list
247 
248  free(last); // Free the old vertices array
249  return drawlist; // Return the display list
250 }
